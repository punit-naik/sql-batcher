001  (ns org.clojars.punit-naik.sql-batcher
002    (:require [clojure.java.jdbc :as db]
003              [clojure.string :as str]
004              [taoensso.timbre :as log]))
005  
006  (defn update-or-delete-op
007    "Checks if the large query is doing an update or delete operation"
008    [query]
009    (if (str/starts-with? query "delete from")
010      :delete :update))
011  
012  (defn table-name
013    "Gets the name of the table from the large update/delete query"
014    [query]
015    (let [query-parts (str/split query #"\s+")]
016      (nth query-parts
017           (if (= :delete (update-or-delete-op query))
018             2 1))))
019  
020  (defn primary-key-column
021    "Assumes the primary key column to be `<table-name>_id`"
022    [query]
023    (str (table-name query) "_id"))
024  
025  (defn remove-trailing-semicolon
026    "Removes semi-colon from the end of a string"
027    [s]
028    (if (str/ends-with? s ";")
029      (str/join (butlast s))
030      s))
031  
032  (defn extract-where-clause
033    "Extracts where clause from the large update/delete query"
034    [query]
035    (->> (str/split query #"where ")
036         second
037         str/trim
038         str/trim-newline
039         (#(str/split % #"and"))
040         (map str/trim)
041         (str/join " and ")))
042  
043  (defn build-select-query
044    [query pkey-column]
045    (str "select " pkey-column
046         " from " (table-name query)
047         " where " (extract-where-clause query)
048         " order by " pkey-column
049         " limit ?,?"))
050  
051  (defn get-pkey-batch
052    [db-spec select-query offset limit pkey-column]
053    (db/query
054     db-spec
055     [select-query offset limit]
056     {:row-fn (keyword pkey-column)}))
057  
058  (defn where-in-clause
059    [column-name values]
060    (into [(str column-name
061                " in ("
062                (str/join "," (repeat (count values) "?"))
063                ")")]
064          values))
065  
066  (def set-clause
067    "Gets the set column names and their values from the update query"
068    (memoize
069     (fn [query]
070       (-> query
071           (str/split #" where ")
072           first
073           (str/split #"set ")
074           second
075           str/trim
076           str/trim-newline
077           (str/split #",")
078           (->> (map
079                 (fn [s]
080                   (let [[col-name value] (str/split s #"\=")
081                         value (->> value str/trim-newline str/trim)]
082                     [(->> col-name str/trim-newline str/trim keyword)
083                      (if (re-matches #"[0-9]+" value) (read-string value) value)])))
084                (into {}))))))
085  
086  (defn execute!
087    "Executes the update/delete query
088     Returns the number of rows processed"
089    [db-spec query pkey-column pkey-batch]
090    (when (seq pkey-batch)
091      (log/info (str "Executing query " query " for a batch of size " (count pkey-batch)))
092      (let [table (keyword (table-name query))
093            where (where-in-clause pkey-column pkey-batch)]
094        (first
095         (if (= :update (update-or-delete-op query))
096           (db/update! db-spec table (set-clause query) where)
097           (db/delete! db-spec table where))))))
098  
099  (defn start-batch-processing!
100    [db-spec query batch-size & [pkey-column]]
101    (log/info "Starting the batch processing of large query:" query)
102    (let [pkey-column (str/lower-case (or pkey-column (primary-key-column query)))
103          query (-> query str/lower-case remove-trailing-semicolon)
104          select-query (build-select-query query pkey-column)
105          delete-op? (= :delete (update-or-delete-op query))]
106      (loop [offset 0
107             total-rows-processed 0
108             finished? false]
109        (if finished?
110          (do (log/info "Total number of rows processed:" total-rows-processed)
111              (log/info "Finished batch processing of query:" query))
112          (let [batch (get-pkey-batch db-spec select-query offset batch-size pkey-column)
113                batch-count (count batch)]
114            (recur (if delete-op? offset (+ offset batch-count))
115                   (+ total-rows-processed (or (execute! db-spec query pkey-column batch) 0))
116                   (zero? batch-count)))))))
